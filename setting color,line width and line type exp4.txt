import matplotlib.pyplot as plt

# Sample Data
x = [1, 2, 3, 4, 5]
y = [2, 3, 5, 7, 11]

# Create a Customized Line Plot
plt.plot(x, y, color='orange', linewidth=3, linestyle='--')

# Add Labels and Title
plt.xlabel('X-Axis')
plt.ylabel('Y-Axis')
plt.title('Customized Line Plot')

# Show the plot
plt.show()
